extend tool

关于extend的使用方法

```bash

    /*jQuery的extend
     * 如果传入多个参数，那么把后面的对象混入给第一个成员
     * 如果传入1个参数，谁调用就是给谁copy
     * 
     */
    var a = {age :18};
    var b = {name:'四哥'};
    var c = {hobby:'逛吃'};
    $.extend(a,b,c);//就是把b,c都混入给a 相当于copy
    console.log(a);//Object {age: 18, name: "sige", hobby: "chi"}
    //给$添加静态成员
    $.extend(a)//把a混入$
    console.log($);
    $.fn.extend({
        con:function(msg){
            console.log(msg)
        }
    })
    $().con('打印试试');

    function extend(o1,o2){
        //混入只混入自己不混入继承过来的，这边做一个判断，判断是不是自己的hasOwnProperty
        for(var k in o2){
            if(o2.hasOwnProperty(k)){
                   o1[k] = o2[k];
            }

        }
    }
    //函数式子
    function extend1(obj){
        for(var key in obj){
            if(obj.hasOwnProperty(key)){
                this[key] = obj[key];
            }
        }
    }
    //比如
    var o = {
        a:1,
        ext:extend1
    }
    o.ext({
        val:'我是混入的值'
    })
    console.log(o);
